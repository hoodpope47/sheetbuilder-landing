type NavItem = {

    icon: React.ComponentType<React.SVGProps<SVGSVGElement>>;
};

    {
        label: "Overview",

function isActivePath(pathname: string, href: string) {
    if (href === "/dashboard") return pathname === "/dashboard";
    return pathname.startsWith(href);
}
